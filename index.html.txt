            <label for="purpleBoxes">Number of Purple Resource Choice Boxes:</label>
            <input type="number" id="purpleBoxes" name="purpleBoxes" min="0" required>

            <label for="goldBoxes">Number of Gold Resource Boxes:</label>
            <input type="number" id="goldBoxes" name="goldBoxes" min="0" required>

            <label for="woodMillions">Amount of Wood (in millions):</label>
            <input type="number" id="woodMillions" name="woodMillions" min="0" required>

            <label for="ironMillions">Amount of Iron (in millions):</label>
            <input type="number" id="ironMillions" name="ironMillions" min="0" required>

            <label for="electricityMillions">Amount of Electricity (in millions):</label>
            <input type="number" id="electricityMillions" name="electricityMillions" min="0" required>

            <label for="requiredForUpgrade">Amount of Each Resource Required for Upgrade:</label>
            <input type="number" id="requiredForUpgrade" name="requiredForUpgrade" min="0" required>

            <button type="submit">Calculate</button>
        </form>
        <div id="result"></div>
    </div>

    <script>
        document.getElementById('resourceForm').addEventListener('submit', function(event) {
            event.preventDefault(); // Prevent default form submission

            // Get input values
            const blueBoxes = parseFloat(document.getElementById('blueBoxes').value);
            const purpleBoxes = parseFloat(document.getElementById('purpleBoxes').value);
            const goldBoxes = parseFloat(document.getElementById('goldBoxes').value);
            const woodMillions = parseFloat(document.getElementById('woodMillions').value);
            const ironMillions = parseFloat(document.getElementById('ironMillions').value);
            const electricityMillions = parseFloat(document.getElementById('electricityMillions').value);
            const requiredForUpgrade = parseFloat(document.getElementById('requiredForUpgrade').value);

            // Perform calculations
            const box_total = (blueBoxes * 10000) + (purpleBoxes * 100000) + (goldBoxes * 1000000);
            const resource_total = woodMillions + ironMillions + electricityMillions; // Already in millions, so no further multiplication needed for millions
            const required_total = requiredForUpgrade * 3;

            const final_answer = required_total - resource_total - box_total;

            // Display the result
            const resultDiv = document.getElementById('result');
            resultDiv.textContent = `The final answer is: ${final_answer.toFixed(2)}`; // .toFixed(2) for better readability
            resultDiv.style.display = 'block'; // Make the result visible
        });
    </script>
